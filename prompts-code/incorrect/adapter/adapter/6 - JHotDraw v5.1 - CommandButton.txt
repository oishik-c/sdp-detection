Given the following java code, determine whether it contains an instance of the adapter design pattern playing the role of adapter.

package CH.ifa.draw.util;

import java.awt.Button;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;



public  class CommandButton
		extends Button implements ActionListener {

	private Command   fCommand;

	
	public CommandButton(Command command) {
		super(command.name());
		fCommand = command;
		addActionListener(this);
	}

	
	public void actionPerformed(ActionEvent e) {
		fCommand.execute();
		if (!getLabel().equals(fCommand.name()) ) {
			setLabel(fCommand.name());
		}
	}
}
